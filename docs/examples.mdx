# Exemples de code

Cette section fournit des exemples de code pour interagir avec notre API dans différents langages de programmation.

## Récupérer des ressources

<CodeTabs>
    <TabsList>
        <TabsTrigger value="javascript">JavaScript</TabsTrigger>
        <TabsTrigger value="python">Python</TabsTrigger>
        <TabsTrigger value="curl">cURL</TabsTrigger>
    </TabsList>
    <TabsContent value="javascript">
        <CodeBlock
            language="JavaScript"
            code={`const apiKey = 'votre_clé_api';

            async function getResources() {
              const response = await fetch('https://api.example.com/v1/resources', {
                method: 'GET',
                headers: {
                  'Authorization': \`Bearer \${apiKey}\`,
                  'Content-Type': 'application/json'
                }
              });

                  return response.json();
                }

                getResources()
                  .then(data => console.log(data))
                  .catch(error => console.error('Erreur:', error));
`}
        />
    </TabsContent>
    <TabsContent value="python">
        <CodeBlock
            language="Python"
            code={`import requests

            api_key = 'votre_clé_api'

            def get_resources():
              headers = {
                'Authorization': f'Bearer {api_key}',
                'Content-Type': 'application/json'
              }

              response = requests.get('[https://api.example.com/v1/resources](https://api.example.com/v1/resources)', headers=headers)
              return response.json()

            print(get_resources())`}
        />
    </TabsContent>
    <TabsContent value="curl">
        <CodeBlock
            language="cURL"
            code={`curl -X GET \
  -H "Authorization: Bearer votre_clé_api" \
  -H "Content-Type: application/json" \
  [https://api.example.com/v1/resources](https://api.example.com/v1/resources)`}
        />
    </TabsContent>
</CodeTabs>

## Créer une ressource

<CodeTabs>
    <TabsList>
        <TabsTrigger value="javascript">JavaScript</TabsTrigger>
        <TabsTrigger value="python">Python</TabsTrigger>
        <TabsTrigger value="curl">cURL</TabsTrigger>
    </TabsList>
    <TabsContent value="javascript">
        <CodeBlock
            language="JavaScript"
            code={`const apiKey = 'votre_clé_api';

                async function createResource() {
                const response = await fetch('[https://api.example.com/v1/resources](https://api.example.com/v1/resources)', {
                method: 'POST',
                headers: {
                'Authorization': \`Bearer \${apiKey}\`,
                  'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                  name: 'Nouvelle ressource',
                  description: 'Description de la nouvelle ressource'
                })`}
        />
    </TabsContent>
    <TabsContent value="python">
        <CodeBlock
            language="Python"
            code={`import requests

            api_key = 'votre_clé_api'

                def create_resource():
                  headers = {
                    'Authorization': f'Bearer {api_key}',
                    'Content-Type': 'application/json'
                  }

                  data = {
                    'name': 'Nouvelle ressource',
                    'description': 'Description de la nouvelle ressource'
                  }

                  response = requests.post('[https://api.example.com/v1/resources](https://api.example.com/v1/resources)', headers=headers, json=data)
                  return response.json()

                print(create_resource())`}
        />
    </TabsContent>
    <TabsContent value="curl">
        <CodeBlock
            language="cURL"
            code={`curl -X POST \
  -H "Authorization: Bearer votre_clé_api" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Nouvelle ressource",
    "description": "Description de la nouvelle ressource"
  }' \
  [https://api.example.com/v1/resources](https://api.example.com/v1/resources)`}
        />
    </TabsContent>
</CodeTabs>
